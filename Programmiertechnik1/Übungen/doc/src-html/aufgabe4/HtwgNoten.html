001package aufgabe4;
002
003
004public final class HtwgNoten {
005    public HtwgNoten () { }
006
007    public double BESTE = 50;
008    public double SCHLECHTESTE = 10;
009    int counter = 0;
010    double durchfall = 0;
011    double sum = 0;
012    int a = 0;
013    int b = 0;
014
015    public boolean istZulaessig (String note) {
016
017        double c = 0;
018        boolean gueltigeNote = false;
019
020        try {
021            boolean  first = Character.isDigit(note.charAt(0));
022            boolean  last  = Character.isDigit(note.charAt(2));
023            char     mitte = note.charAt(1);
024            a = Character.getNumericValue(note.charAt(0));
025            b = Character.getNumericValue(note.charAt(2));
026
027            if (note.length() == 3 && (mitte == '.' || mitte == ',')
028                    && (first && last)) {
029                if (a > 0 && a < 6) {
030                    if ((a == 4 || a == 5) && (b > 0)) {
031                        System.out.println("Unzulaessige Note " + note
032                                           + " wird ignoriert!");
033                    } else {
034
035                        c  = (a * 10) + b;
036                        counter++;
037                        gueltigeNote = true;
038                    }
039                }
040            } else {
041                System.out.println("Unzulaessige Note " + note
042                                   + " wird ignoriert!");
043            }
044        } catch (java.lang.StringIndexOutOfBoundsException x) {
045            System.out.println("Unzulaessige Note " + note
046                               + " wird ignoriert!");
047        }
048        if (gueltigeNote) {
049            if (c < BESTE && counter > 0) {
050                BESTE =  c;
051            }
052            if (c > SCHLECHTESTE) {
053                SCHLECHTESTE =  c;
054            }
055            if (c == 50) {
056                durchfall++;
057            }
058            sum += c;
059        }
060        return false;
061
062    }
063
064
065    public double doubleNote (String note) {
066        Double n = 0.0;
067        note = note.replace(',', '.');
068        try {
069            if(istZulaessig(note)) {
070                n = Double.parseDouble(note);
071            }
072
073        } catch (java.lang.IllegalArgumentException x) {
074            System.out.println("Unzulaessige Note " + note + " wird ignoriert!");
075        }
076        return n;
077    }
078
079    public boolean istBestanden (double bestanden) {
080        String str = String.valueOf(bestanden);
081        if(istZulaessig(str)) {
082            if(bestanden <= 4.0) {
083                return true;
084            } else {
085                return false;
086            }
087
088        }
089        return false;
090    }
091
092    public double bessere (double n, double m) {
093        if(n < m) {
094            return n;
095        }
096        return m;
097
098    }
099
100    public double schlechtere (double n, double m) {
101        if(n > m) {
102            return n;
103        }
104        return m;
105
106    }
107}
