004import java.util.Scanner;
005
006/**
007 * Histogramm liest ganze Zahlen zwischen 1 und 6 ein und
008 * gibt deren H&auml;ufigkeitsverteilung als Histogramm aus.
009 * @author : Abusaleh
010 * @version : 13.04.15
011 */
012public final class Histogramm {
013    private Histogramm() { }
014
015    private static final Scanner EINGABE = new Scanner(System.in);
016
017    /**
018     * main ist der Startpunkt des Programms.
019     * @param args wird nicht verwendet.
020     */
021    public static void main(final String[] args) {
022
023        /*  (1) hier ein Feld von Zaehlern definieren */
024       // final int max = 6;
025       // final int mod = 5;
026        int[] zaehler = {0, 0, 0, 0, 0, 0};
027     //-------------------------------------------------- Zahlen einlesen
028        System.out.println("Ganze Zahlen zwischen 1 und 6 eingeben "
029                    + "(Ende mit Strg-D):");
030
031        while (EINGABE.hasNext()) {
032           
033            int number = EINGABE.nextInt();
034           
035         /*  (2) hier Anweisungen fuer das
036             Pruefen und Zaehlen der Eingabe schreiben */
037            if (number < 1 || number > 6) {
038                System.out.println("Falsche Eingabe wird ignoriert: " + number);
039            } else {
040                
041                ++zaehler[number - 1];
042            }
043        }
044       /*
045         for (int i=0; i<zaehler.length; i++) {
046              System.out.println("Index des Arrays = " + i);
047              System.out.println("Wert unter dem oberen Index = " + zaehler[i]);
048          }
049       */
050        //---------------------------------------------- Histogramm ausgeben
051        /*  (3) hier Anweisungen fuer die Histogrammausgabe schreiben */
052//      int     n = 50/50+1;
053            
054for (int i = 0; i < zaehler.length; ++i) {
055            for (int j = 0; j < zaehler[i]; ++j) {
056                
057                
058                 if (((j + 1) % 5) == 0) {
059                    System.out.print("$");
060                } else  {
061                
062                    System.out.print(i + 1);
063                
064                }
065                
066                
067            
068                }
069            if (zaehler[i] != 0) {
070                System.out.printf(" (%d)%n", zaehler[i]);
071            }
072        }
073
074
075    }
076}
