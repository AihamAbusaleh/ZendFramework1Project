003package einstieg;
004
005import java.util.Scanner;
006import static java.lang.System.out;
007
008/**
009* Einstieg ist ein kleines Java Programm.
010* @author Aiham
011* @version 21.03.15
012*/
013public final class Einstieg {
014    private Einstieg() { }
015    private static final Scanner EINGABE = new Scanner(System.in);
016    /**
017    * Die Klassenmethode main ist der Startpunkt des Programms.
018    * main verwendet die Methoden print und printf zum A usgeben von Text
019    * sowie die Methoden next und nextInt zum Einlesen v on Text
020    * und einer ganzen Zahl.
021    * @param args wird nicht verwendet
022    */
023    public static void main(final String[] args) {
024        out.print("vorname: ");
025        String vorname = EINGABE.next();
026        out.print("Anzahl bisher geschriebener Java-Programm: ");
027        int anzahl = EINGABE.nextInt();
028        out.printf("%s's %d. Java-Programm funktioniert!%n",
029                   vorname, anzahl + 1);
030    }
031}
