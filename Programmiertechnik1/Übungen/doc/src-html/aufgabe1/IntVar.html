009/**
010 * IntVar zeigt den Umgang mit Variablen vom Typ int.
011 * &Uuml;bungsaufgabe 1 zur Programmiertechnik 1.
012 * @author Abusaleh.
013 *@version IntVar.
014*/
015public final class IntVar {
016    private IntVar() { }
017
018    private static final Scanner EINGABE = new Scanner(System.in);
019    /**
020     * main ist der Startpunkt des Programms.
021     * @param args private static final wird nicht verwendet.
022    */
023    public static void main(final String[] args) {
024        /* to-do: Konstanten min und max definieren */
025        int max = Integer.MAX_VALUE;
026        int min = Integer.MIN_VALUE;
027        /* to-do: Eingabeaufforderung ausgeben */
028        System.out.println("Bitte geben Sie die 1.Zahl ein : ");
029        if (!EINGABE.hasNextInt()) {
030            System.out.println("Eingabe ist ungültig");
031        }
032        int zahl1 = EINGABE.nextInt();
033        System.out.println("Bitte geben Sie die 2.Zahl ein : ");
034        if (!EINGABE.hasNextInt()) {
035            System.out.println("Eingabe ist ungültig");
036        }
037        int zahl2 = EINGABE.nextInt();
038        long z1 = zahl1;
039        long z2 = zahl2;
040        /* to-do: zwei ganze Zahlen einlesen */
041        System.out.printf("Die eingegebenen Integer Zahlen : %d und %d %n ",
042            zahl1, zahl2);
043        /* to-do: die beiden Zahlen dezimal, okatal und hexadezimal ausgeben */
044        System.out.printf(" Zahlen in Dezimal %d und  %d %n", zahl1, zahl2);
045        System.out.printf(" Zahlen in Oktal %o und  %o %n", zahl1, zahl2);
046        System.out.printf(" Zahlen in Hexa %x und  %x %n", zahl1, zahl2);
047        /* to-do: alle zweistelligen arithmetischen Operatoren ausprobieren */
048        System.out.println("Bitte geben Sie den gewünschten Operator ein  : ");
049        String verrechnen = EINGABE.next();
050        String plus = "+";
051        String minus = "-";
052        String mult = "*";
053        String div = "/";
054        String lt = "<";
055        String gt = ">";
056        String gl = "==";
057        String ug = "!=";
058        switch(verrechnen) {
059        case "+":
060            System.out.println(zahl1 + zahl2);
061            break;
062        case "-":
063            System.out.println(zahl1 - zahl2);
064            break;
065        case "*":
066            System.out.println(zahl1 * zahl2);
067            break;
068        case "/":
069            System.out.println(zahl1 / zahl2);
070            break;
071        case "<":
072            System.out.println(zahl1 < zahl2);
073            break;
074        case ">":
075            System.out.println(zahl1 > zahl2);
076            break;
077        case "==":
078            System.out.println(zahl1 == zahl2);
079            break;
080        case "!=":
081            System.out.println(zahl1 != zahl2);
082            break;
083        default:
084            System.out.println("Illegal operator !");
085            break;
086        }
087        long l = z1 * z2;
088        System.out.println("Zahl1 * Zahl2 = " + l);
089        if (l < min || l > max) {
090            System.out.println("Ergebnis kann NICHT als Int dargestellt werdn");
091        } else {
092            System.out.println("Ergebnis kann als Int dargestellt werden" + l);
093        }
094    }
095}
