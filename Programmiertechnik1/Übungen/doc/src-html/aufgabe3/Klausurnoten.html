003package aufgabe3;
004
005import java.util.Locale;
006import java.util.Scanner;
007
008/**
009 * Klausurnoten erstellt eine Notenstatistik.
010 * <p>
011 * Das Programm erwartet Noten im Format
012 * <tt>Ganze,Zehntel</tt> oder <tt>Ganze.Zehntel</tt>,
013 * wobei <tt>Ganze</tt> und <tt>Zehntel</tt> nur aus
014 * je einer Dezimalziffer bestehen d&uuml;rfen.
015 * Andere Eingaben werden wegen Formatfehler ignoriert.
016 * </p>
017 * <p>
018 * Das Programm gibt die folgende Statistik aus:
019 * </p>
020 * <ul>
021 * <li>die Anzahl der ber&uuml;cksichtigten Noten</li>
022 * <li>die beste Note</li>
023 * <li>die schlechteste Note</li>
024 * <li>den Durchschnitt</li>
025 * <li>die Durchfallquote in Prozent</li>
026 * </ul>
027 * <p>
028 * Es werden in der Statistik nur die nach HTWG-Pr&uuml;fungsordnung
029 * zul&auml;ssigen Noten  (1,0 1,3 1,7 2,0 2,3 2,7 3,0 3,3 3,7 4,0 5,0)
030* ber&uuml;cksichtigt.
031 * Andere Eingaben werden wegen falscher Vorkommastelle oder
032 * falscher Nachkommastelle ignoriert.
033 * Alle Noten bis 4,0 gelten als bestanden, nur die 5,0 als durchgefallen.
034 * </p>
035 *
036 * @author Abusaleh
037 * @version 27.04.2015
038 */
039public final class Klausurnoten {
040    private Klausurnoten() { }
041    private static final Scanner EINGABE = new Scanner(System.in);
042
043    /**
044     * main ist der Startpunkt des Programms.
045     * @param args wird nicht verwendet.
046     */
047    public static void main(final String[] args) {
048        Locale.setDefault(Locale.GERMANY);
049        System.out.println("Noten im Format Ganze,Zehntel "
050                           + "oder Ganze.Zehntel eingeben (Ende mit Strg-D):");
051        int counter = 0;
052        int durchfall = 0;
053        int sum = 0;
054        int a = 0;
055        int b = 0;
056        int schlechteNote = 10;
057        int  besteNote     = 50;
058        while (EINGABE.hasNext()) {
059            String note = EINGABE.next();
060            int c = 0;
061            boolean gueltigeNote = false;
062            try {
063                boolean  first = Character.isDigit(note.charAt(0));
064                boolean  last  = Character.isDigit(note.charAt(2));
065                char     mitte = note.charAt(1);
066                a = Character.getNumericValue(note.charAt(0));
067                b = Character.getNumericValue(note.charAt(2));
068
069                if (note.length() == 3 && (mitte == '.' || mitte == ',')
070                    && (first && last)) {
071                    if (a > 0 && a < 6) {
072                        switch (note.charAt(2)) {
073                        case '0':
074                            c  = (a * 10) + b;
075                            counter++;
076                            gueltigeNote = true;
077                            break;
078                        case '3':
079                        case '7':
080                            if (note.charAt(0) == '4'
081                                || note.charAt(0) == '5') {
082                                System.out.println("Note " + note
083                                    + " wird wegen Nachkommastelle "
084                                    + note.charAt(2) + " ignoriert!");
085                                break;
086                            }
087                            c  = (a * 10) + b;
088                            counter++;
089                            gueltigeNote = true;
090                            break;
091                        default:
092                            System.out.println("Note " + note
093                                + " wird wegen Nachkommastelle "
094                                + note.charAt(2) + " ignoriert!");
095                            break;
096                        }
097                    } else {
098                        System.out.println("Note " + note
099                            + " wird wegen Vorkommastelle "
100                            + note.charAt(0) + " ignoriert!");
101                    }
102                } else {
103                    System.out.println("Note " + note
104                        + " wird wegen Formatfehler ignoriert!");
105                }
106            } catch (java.lang.StringIndexOutOfBoundsException x) {
107                System.out.println("Note " + note
108                    + " wird wegen Formatfehler ignoriert!");
109            }
110            if (gueltigeNote) {
111                if (c < besteNote && counter > 0) {
112                    besteNote =  c;
113                }
114                if (c > schlechteNote) {
115                    schlechteNote =  c;
116                }
117                if (c == 50) {
118                    durchfall++;
119                }
120                sum += c;
121            }
122
123        }
124
125        System.out.println();
126        System.out.printf("Anzahl beruecksichtigter Noten: %d%n", counter);
127        double beste = besteNote / 10.0;
128        double schlechteste = schlechteNote / 10.0;
129        double durchfallquote = 0;
130        if (counter != 0) {
131            System.out.printf("Beste Note: %.1f " , beste);
132            System.out.printf("%nSchlechteste Note: %.1f" , schlechteste);
133            double durchschnitt = (double) sum / counter;
134            System.out.printf("%nDurchschnitt: %.1f " , durchschnitt / 10);
135            durchfallquote =  (durchfall * 100.0 / counter);
136            System.out.printf("%nDurchfallquote: %.1f%% ",
137                              (durchfallquote));
138            System.out.println();
139        }
140    }
141}
